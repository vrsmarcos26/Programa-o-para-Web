            // Função para pegar os parâmetros da URL
        function getParametro(url, nomeParametro) {
            const parametros = new URLSearchParams(url.search);
            return parametros.get(nomeParametro);
        }

        // Pegando o parâmetro 'pre' da URL
        const urlAtual = window.location;
        const valorPre = getParametro(urlAtual, 'pre');

        // Exibindo o valor do parâmetro na página
        const resultado = document.getElementById('resultado');
        if (valorPre) {
            resultado.innerHTML = `O valor do parâmetro "pre" é: ${valorPre}`;
        } else {
            resultado.innerHTML = 'Parâmetro "pre" não encontrado na URL.';
        }
